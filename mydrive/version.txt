pipeline {
    agent any

    stages {
        stage('Checkout Code') {
            steps {
                git credentialsId: 'bitbucket-creds-id',
                    url: 'https://bitbucket.org/your-workspace/your-repo.git',
                    branch: 'main'
            }
        }

        stage('Print Working Directory & List Files') {
            steps {
                sh '''
                    echo "Current directory:"
                    pwd
                    echo "Contents:"
                    ls -al
           
            }
        }
    }
}
