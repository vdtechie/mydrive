pipeline {
    agent { label 'python3-node' } 

    stages {
        stage('Checkout Code') {
            steps {
                git credentialsId: 'bitbucket-creds-id',
                    url: 'https://bitbucket.org/your-workspace/your-repo.git',
                    branch: 'main'
            }
        }

        stage('Print Working Directory & List Files') {
            steps {
                sh '''
                    echo "Current directory:"
                    pwd
                    echo "Contents:"
                    ls -al
           
            }
        }

        stage('Environment Info') {
            steps {
                sh '''
                    echo "Python version:"
                    python3 --version
                    pwd
                    ls -al
                '''
                }
            }
        }
}
